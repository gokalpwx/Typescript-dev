type DeepReadonly<T> = {
  readonly [K in keyof T]:
    T[K] extends (infer U)[] ? ReadonlyArray<DeepReadonly<U>> :
    T[K] extends object ? DeepReadonly<T[K]> :
    T[K];
};


type Config = { api: { url: string }, features: string[] };
const config: DeepReadonly<Config> = { api: { url: "http://..." }, features: ["A","B"] };

config.api.url = "yeni-url";

config.features.push("C");
console.log("Soru 14 Başarılı!", config.api.url);
